SET LINESIZE 130;
SET PAGESIZE 40;
TTITLE CENTER 'Report 1: Maximum City Loan Demand per Quarter' SKIP 2

COLUMN CAL_YEAR FORMAT 9999 HEADING 'Year'
COLUMN CAL_QUARTER FORMAT 9 HEADING 'Qtr'
COLUMN MEMBER_CITY FORMAT A20 HEADING 'City'
COLUMN TOTAL_LOANS FORMAT 999,999 HEADING 'Total Loans'
COLUMN LOANS_PER_MEMBER FORMAT 999.99 HEADING 'Loans/Member'
COLUMN PCT_OF_QTR FORMAT 999.99 HEADING '%ofQtr'

WITH loan_stats AS (
    SELECT 
        D.CAL_YEAR,
        D.CAL_QUARTER,
        M.MEMBER_CITY,
        COUNT(L.LOAN_DATE) AS total_loans,
        ROUND(COUNT(L.LOAN_DATE) / COUNT(DISTINCT L.MEMBER_KEY), 2) AS loans_per_member,
        ROW_NUMBER() OVER (
            PARTITION BY D.CAL_YEAR, D.CAL_QUARTER
            ORDER BY COUNT(L.LOAN_DATE) DESC
        ) AS rn,
        SUM(COUNT(L.LOAN_DATE)) OVER (PARTITION BY D.CAL_YEAR, D.CAL_QUARTER) AS qtr_total
    FROM BOOK_LOAN_FACT L
    JOIN DATE_DIM D ON L.DATE_KEY = D.DATE_KEY
    JOIN MEMBER_DIM M ON L.MEMBER_KEY = M.MEMBER_KEY
    GROUP BY D.CAL_YEAR, D.CAL_QUARTER, M.MEMBER_CITY
)
SELECT 
    CAL_YEAR,
    CAL_QUARTER,
    MEMBER_CITY,
    total_loans,
    loans_per_member,
    ROUND((total_loans / qtr_total) * 100, 2) AS pct_of_qtr
FROM loan_stats
WHERE rn = 1
ORDER BY CAL_YEAR DESC, CAL_QUARTER;
