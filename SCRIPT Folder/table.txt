/*
DROP TABLE BOOK CASCADE CONSTRAINTS;
DROP TABLE MEMBER CASCADE CONSTRAINTS;
DROP TABLE FACILITY CASCADE CONSTRAINTS;
DROP TABLE LOAN_RECORD CASCADE CONSTRAINTS;
DROP TABLE FINE_INVOICES CASCADE CONSTRAINTS;
DROP TABLE FACILITY_BOOKING CASCADE CONSTRAINTS;
*/

CREATE TABLE BOOK (
    BOOK_ID               VARCHAR2(5)  NOT NULL CHECK(REGEXP_LIKE(BOOK_ID, '^B[0-9]{4}$')), 
    TITLE                 VARCHAR2(80)  NOT NULL,
    AUTHOR                VARCHAR2(60),
    PUBLISHER             VARCHAR2(60), 
    PUBLISHED_YEAR        NUMBER(4),
    GENRE                 VARCHAR2(30)  NOT NULL, 
    LANGUAGE              VARCHAR2(20)  NOT NULL,
    PRICE                 NUMBER(8,2),
    POPULARITY_SCORE      NUMBER CHECK(POPULARITY_SCORE >=0 AND POPULARITY_SCORE <= 10),
    PRIMARY KEY(BOOK_ID)
);

CREATE TABLE MEMBER (
    MEMBER_ID          VARCHAR2(5)  NOT NULL CHECK(REGEXP_LIKE(MEMBER_ID, '^M[0-9]{4}$')),
    MEMBER_NAME        VARCHAR2(60)  NOT NULL,      
    MEMBER_EMAIL       VARCHAR2(80) UNIQUE NOT NULL,
    MEMBER_PHONE_NUM   VARCHAR2(20)  NOT NULL,
    MEMBER_ADDRESS     VARCHAR2(100),
    MEMBER_CITY        VARCHAR2(40),
    MEMBER_COUNTRY     VARCHAR2(45),
    MEMBER_STATE       VARCHAR2(30),
    MEMBER_ZIP_CODE    VARCHAR2(10),
    MEMBER_START_DATE  DATE,
    MEMBER_END_DATE   DATE,
    IS_ACTIVE          VARCHAR2(1) DEFAULT 'Y' CHECK(IS_ACTIVE IN('Y','N')),
    PRIMARY KEY(MEMBER_ID)
);

CREATE TABLE FACILITY (
    FACILITY_ID           VARCHAR2(5)  NOT NULL CHECK(REGEXP_LIKE(FACILITY_ID, '^F[0-9]{4}$')),        
    FACILITY_NAME         VARCHAR2(40),
    FACILITY_TYPE         VARCHAR2(40),
    CAPACITY              NUMBER,
    BOOKING_FEE_PER_HOUR  NUMBER(6,2),
    PRIMARY KEY(FACILITY_ID)
);


CREATE TABLE LOAN_RECORD (
    LOAN_ID          VARCHAR2(5)  NOT NULL CHECK(REGEXP_LIKE(LOAN_ID, '^L[0-9]{4}$')),
    BOOK_ID          VARCHAR2(5)  NOT NULL,
    MEMBER_ID        VARCHAR2(5)  NOT NULL,
    LOAN_DATE        DATE NOT NULL,
    DUE_DATE         DATE NOT NULL,
    RETURN_DATE      DATE NOT NULL,
    IS_EXTENDED      VARCHAR2(1)  DEFAULT 'N' CHECK(IS_EXTENDED IN('Y','N')),
    EXTENSION_COUNT  NUMBER,
    PRIMARY KEY(LOAN_ID),
    CONSTRAINT FK_BOOK_ID FOREIGN KEY (BOOK_ID) REFERENCES BOOK(BOOK_ID),
    CONSTRAINT FK_MEMBER_ID FOREIGN KEY (MEMBER_ID) REFERENCES MEMBER(MEMBER_ID)  
);

CREATE TABLE FINE_INVOICES (
    FINE_ID       VARCHAR2(6) NOT NULL CHECK(REGEXP_LIKE(FINE_ID, '^FI[0-9]{4}$')),
    LOAN_ID       VARCHAR2(5) NOT NULL,
    OVERDUE_DAYS  NUMBER,
    FINE_PER_DAY  NUMBER(5,2),
    FINE_AMT      NUMBER(8,2),
    PRIMARY KEY(FINE_ID),
    CONSTRAINT FK_LOAN_ID FOREIGN KEY (LOAN_ID) REFERENCES LOAN_RECORD(LOAN_ID)
);

CREATE TABLE FACILITY_BOOKING (
    BOOKING_ID          VARCHAR2(6) NOT NULL CHECK(REGEXP_LIKE(BOOKING_ID, '^BK[0-9]{4}$')),
    MEMBER_ID           VARCHAR2(5) NOT NULL,
    FACILITY_ID         VARCHAR2(5) NOT NULL,
    BOOKING_DATE        DATE,
    BOOKING_START_TIME  DATE,
    BOOKING_END_TIME    DATE,
    BOOKING_FEE_AMT     NUMBER(8,2),
    IS_CHECKED_IN       VARCHAR2(1)  DEFAULT 'Y' CHECK(IS_CHECKED_IN IN('Y','N')),
    PRIMARY KEY(BOOKING_ID),
    CONSTRAINT FK_FB_MEMBER_ID FOREIGN KEY (MEMBER_ID) REFERENCES MEMBER(MEMBER_ID),
    CONSTRAINT FK_FACILITY_ID FOREIGN KEY (FACILITY_ID) REFERENCES FACILITY(FACILITY_ID)
);
