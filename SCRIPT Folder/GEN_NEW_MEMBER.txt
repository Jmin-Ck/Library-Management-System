-- CREATE NEW_CUST TABLE TO STORE ALL THE 3840 ROWS OF RECORDS -- UNIQUE
create table NEW_MBR
(
 MEMBER_ID          NUMBER NOT NULL,
 MEMBER_NAME        VARCHAR2(60)  NOT NULL,      
 MEMBER_EMAIL       VARCHAR2(80) UNIQUE NOT NULL,
 MEMBER_PHONE_NUM   VARCHAR2(20)  NOT NULL,
 MEMBER_ADDRESS     VARCHAR2(100),
 MEMBER_CITY        VARCHAR2(40),
 MEMBER_COUNTRY     VARCHAR2(45),
 MEMBER_STATE       VARCHAR2(30),
 MEMBER_ZIP_CODE    VARCHAR2(10),
 MEMBER_START_DATE  DATE,
 MEMBER_END_DATE   DATE,
 IS_ACTIVE          VARCHAR2(1) DEFAULT 'Y' CHECK(IS_ACTIVE IN('Y','N')),
 CONSTRAINT PK_MEMBERID_NEW_MBR PRIMARY KEY(MEMBER_ID)
);
