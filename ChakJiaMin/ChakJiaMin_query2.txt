SET LINESIZE 130;
SET PAGESIZE 40;

TTITLE CENTER 'Report 2: Book Loan Patterns and Member Reading Behaviour (Max Loans per Quarter with Country)' SKIP 2

COLUMN MEMBER_SEGMENT FORMAT A25 HEADING 'Mbr_Segment'
COLUMN MEMBER_COUNTRY FORMAT A20 HEADING 'Country'
COLUMN TOTAL_LOANS FORMAT 999,999 HEADING 'Total Loans'
COLUMN TOTAL_FINES FORMAT 999,999 HEADING 'Total Fines'
COLUMN AVG_OVERDUE_DAYS FORMAT 999,999 HEADING 'Avg Overdue Days'
COLUMN CAL_YEAR FORMAT 9999 HEADING 'Year'
COLUMN CAL_QUARTER FORMAT 9 HEADING 'Quarter'
COLUMN GENRE FORMAT A15 HEADING 'Genre'

WITH loan_stats AS (
    SELECT 
        CASE 
            WHEN MONTHS_BETWEEN(SYSDATE, M.MEMBER_START_DATE) > 24 THEN 'Loyal Members'
            WHEN MONTHS_BETWEEN(SYSDATE, M.MEMBER_START_DATE) > 12 THEN 'Regular Members'
            ELSE 'New Members'
        END AS member_segment,
        M.MEMBER_COUNTRY,
        COUNT(L.LOAN_DATE) AS total_loans,
        SUM(L.FINE_AMT) AS total_fines,
        AVG(L.OVERDUE_DAYS) AS avg_overdue_days,
        D.CAL_YEAR,
        D.CAL_QUARTER,
        B.GENRE,
        RANK() OVER (
            PARTITION BY D.CAL_YEAR, D.CAL_QUARTER
            ORDER BY COUNT(L.LOAN_DATE) DESC
        ) AS loan_rank
    FROM BOOK_LOAN_FACT L
    JOIN MEMBER_DIM M ON L.MEMBER_KEY = M.MEMBER_KEY
    JOIN DATE_DIM D ON L.DATE_KEY = D.DATE_KEY
    JOIN BOOK_DIM B ON L.BOOK_KEY = B.BOOK_KEY
    GROUP BY 
        CASE 
            WHEN MONTHS_BETWEEN(SYSDATE, M.MEMBER_START_DATE) > 24 THEN 'Loyal Members'
            WHEN MONTHS_BETWEEN(SYSDATE, M.MEMBER_START_DATE) > 12 THEN 'Regular Members'
            ELSE 'New Members'
        END,
        M.MEMBER_COUNTRY,
        D.CAL_YEAR, D.CAL_QUARTER, B.GENRE
)
SELECT 
    member_segment,
    member_country,
    total_loans,
    total_fines,
    avg_overdue_days,
    CAL_YEAR,
    CAL_QUARTER,
    GENRE
FROM loan_stats
WHERE loan_rank = 1
ORDER BY CAL_YEAR DESC, CAL_QUARTER;
